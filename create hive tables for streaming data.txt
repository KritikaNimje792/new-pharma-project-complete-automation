CREATE DATABASE IF NOT EXISTS pharma_streaming_db;

use pharma_streaming_db;

CREATE TABLE pharma_streaming_db.temperature_streaming (
    sensor_id STRING,
    temperature DOUBLE,
    humidity DOUBLE,
    temperature_c DOUBLE,
    unit STRING,
    event_time TIMESTAMP,
    partition_date DATE
)
USING PARQUET
PARTITIONED BY (partition_date)
LOCATION 's3://pharma-data-lake-kritika/streaming_kinesis/temperature';

CREATE TABLE pharma_streaming_db.sales_streaming (
    transaction_id STRING,
    pharmacy_id STRING,
    quantity_sold INT,
    unit_price DOUBLE,
    discount_pct DOUBLE,
    revenue DOUBLE,
    fulfillment_status STRING,
    event_time TIMESTAMP,
    partition_date DATE
)
USING PARQUET
PARTITIONED BY (partition_date)
LOCATION 's3://pharma-data-lake-kritika/streaming_kinesis/sales';


CREATE TABLE pharma_streaming_db.inventory_streaming (
    event_id STRING,
    distributor_id STRING,
    product_sku STRING,
    batch_id STRING,
    change_type STRING,
    quantity_delta INT,
    current_stock INT,
    event_time TIMESTAMP,
    partition_date DATE
)
USING PARQUET
PARTITIONED BY (partition_date)
LOCATION 's3://pharma-data-lake-kritika/streaming_kinesis/inventory';
